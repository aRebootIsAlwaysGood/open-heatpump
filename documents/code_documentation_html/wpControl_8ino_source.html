151 {
152  static uint32_t starttime; // Vergangene Zeit ab Anforderung um Anlaufphase zu bestimmen
153 
154  switch(wpState)
155  {
156  case WP_STATE_IDLE: // Default case
157  getDIOstates();
158  if (wpReqFunc== (WP_REQ_FUNC_LADEN || WP_REQ_FUNC_DEFROST)){
159  starttime=millis(); // Startzeitpunkt
160  wpState= WP_STATE_START;
161  }
162  reglerStatemachine(REGLER_STATE_AUTO); // rufe Regler auf
163  Systemzustand.vorlaufregler= 1; // update Regler Systemzustandsbit
164  break;
165 
166  case WP_STATE_START: // Startsequenz einleiten, Eingänge prüfen
167 
168  getDIOstates();
169  if ((!DiStates.status_nd) || (!DiStates.status_hd)){// Druckalarm wenn Unter-/Überdruck
170  wpState = WP_STATE_ERROR_P;
171  }
172 
173  else if (!DiStates.status_motprotect){ // Wicklungsschutz Motor ausgelöst
174  wpState = WP_STATE_ERROR_M;
175  }
176 
177  // Mischventil öffnen
178  else if ((millis()-starttime) <= T_MISCHERSTELLZEIT){
179  reglerStatemachine(REGLER_STATE_MANUAL); // Regler IO-Steuerung übernehmen
180  digitalWrite(PIN_MISCHER_AUF, HIGH);
181  digitalWrite(PIN_MISCHER_AUF, LOW);
182  digitalWrite(PIN_HEIZPUMPE, HIGH);
183  }
184 
185  // Anlassen
186  else if ((millis()-starttime) <= (T_ANLASS+ T_MISCHERSTELLZEIT)){
187  digitalWrite(PIN_BYPASS, HIGH); // evt auch LOW (Lastanlauf)
188  digitalWrite(PIN_K_ANLAUF, HIGH);
189  digitalWrite(PIN_K_BETRIEB, LOW);
190  digitalWrite(PIN_SUMPFHEIZUNG, LOW);
191  }
192 
193  // Anlassen komplett
194  else{
195  digitalWrite(PIN_K_ANLAUF, HIGH);
196  digitalWrite(PIN_K_BETRIEB, HIGH);
197  digitalWrite(PIN_BYPASS, LOW);
198  digitalWrite(PIN_MISCHER_AUF,LOW);
199  starttime=millis(); //Reset Laufzeittimer auf RUN
200  // Enteisung angefordert
201  if (wpReqFunc== WP_REQ_FUNC_DEFROST){
202  wpState= WP_STATE_DEFROST;
203  }
204  // Laden, gehe zu RUN
205  else{
206  wpState= WP_STATE_RUN;
207  }
208  }
209 
210  break;
211 
212  case WP_STATE_RUN: // Speicher laden bis Anforderung kommt aufzuhören
213  getDIOstates();
214  if ((!DiStates.status_nd) || (!DiStates.status_hd)){ // Druckcheck
215  wpState = WP_STATE_ERROR_P;
216  }
217 
218  else if (!DiStates.status_motprotect){ // Überstromcheck
219  wpState = WP_STATE_ERROR_M;
220  }
221 
222  else if (wpReqFunc== WP_REQ_FUNC_HALT){ // Stop-Anforderung
223  starttime= millis();
224  wpState= WP_STATE_STOP;
225  }
226  else if (wpReqFunc== WP_REQ_FUNC_DEFROST){ // Enteisungs-Anforderung
227  initRegler();
228  wpState= WP_STATE_DEFROST;
229  }
230 
231  else{
232  digitalWrite(PIN_K_ANLAUF,HIGH);
233  digitalWrite(PIN_K_BETRIEB,HIGH);
234  digitalWrite(PIN_VENTILATOR,HIGH);
235  digitalWrite(PIN_LADEPUMPE,HIGH);
236  digitalWrite(PIN_BYPASS, LOW);
237  reglerStatemachine(REGLER_STATE_AUTO); // Regler hat wieder IO Kontrolle
238  }
239 
240  break;
241 
242  case WP_STATE_STOP: // Beende Betrieb, schalte Sumpfheizung ein, gehe zu IDLE
243  digitalWrite(PIN_K_BETRIEB,LOW);
244  digitalWrite(PIN_VENTILATOR, LOW);
245  // digitalWrite(PIN_BYPASS, HIGH); // Bypass auf für Druckausgleich
246  digitalWrite(PIN_SUMPFHEIZUNG, HIGH);
247 
248  reglerStatemachine(REGLER_STATE_AUTO); // Regler wieder autonom
249 
250  if ((millis()-starttime) >= 500){ // Softstop durch verzögertes AUS Anlaufschütz
251  digitalWrite(PIN_K_ANLAUF,LOW);
252  }
253  if ((millis()-starttime) <= T_NACHLAUF){ // Ladepumpe Nachlauf nur nach Laden
254  digitalWrite(PIN_LADEPUMPE, HIGH);
255  }
256  else{
257  digitalWrite(PIN_LADEPUMPE,LOW);
258  initRegler(); // Regler neu Initialisieren
259  wpState= WP_STATE_IDLE;
260  }
261 
262  break;
263 
264  case WP_STATE_DEFROST: // Enteisen, Ventilator + Ladepumpe aus
265  getDIOstates();
266  if ((!DiStates.status_nd) || (!DiStates.status_hd)){ // Druckcheck
267  wpState = WP_STATE_ERROR_P;
268  }
269

271  wpState = WP_STATE_ERROR_M;
272  }
273

275  wpState= WP_STATE_STOP;
276  }
277

279  wpState= WP_STATE_RUN;
280  }
281

285  digitalWrite(PIN_K_BETRIEB, HIGH);
286  digitalWrite(PIN_K_ANLAUF, HIGH);
287  digitalWrite(PIN_VENTILATOR, HIGH);
288  digitalWrite(PIN_LADEPUMPE, HIGH);
