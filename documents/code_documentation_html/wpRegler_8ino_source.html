193 void tristateRegler(float w,float x,float kp,float e_min, float hyst, float tn){
194  float e= 0;
195  float ep= 0;
196  float yr= 0;
197  static float ty;
198  static float ep_old;
199  static uint32_t lastcalctime; // Zeitpunkt letzte Berechnung
200 
201  // set w & x to 0 will reinitialize the controller to zero
202  if(w==0 && x==0){
203  ty=0;
204  ep_old=0;
205  lastcalctime=0;
206  digitalWrite(PIN_MISCHER_ZU, LOW);
207  digitalWrite(PIN_MISCHER_AUF, LOW);
208  }
209 
210  // normal controller loop
211  else if((millis()-lastcalctime)>(tn*1000)){
212  e= w-x; // berechne Regelabweichung
213  yr= (kp*ty)/(ty+tn); // berechne Rückführwert. kp und tn konstant!
214  ep= e-yr; // berechne Prognosewert Regelabweichung mittels Rückführwert
215 
216  // Hysterese Totzonengrenze (Ansprechempfindlichkeit e_min verändern)
217  if(abs(ep)>abs(ep_old)){e_min+=hyst;} //Regelabw steigt->e_min vergr
218  else if (abs(ep)<abs(ep_old)) {e_min-=hyst;} //Abwsinkt->e_min verkl.
219 
220  // Regelabweichung ausserhalb Totzone/2 (e_min) und Hysterese
221  if(abs(ep)>e_min){
222  ty += tn; // zähle Laufzeit hoch
223  // öffne oder schliesse Mischventil
224  if(ep< 0){digitalWrite(PIN_MISCHER_ZU, HIGH);} // wenn Abw negativ
225  else{digitalWrite(PIN_MISCHER_AUF, HIGH);} // Abw positiv
226  }
227  // Regelabw innerhalb Totzone, Ausgang 0
228  else{
229  digitalWrite(PIN_MISCHER_ZU, LOW);
230  digitalWrite(PIN_MISCHER_AUF, LOW);
231  if(ty > 0){ty -=tn;} // Ziehe Laufzeit ab
232  }
233  ep_old=ep; // speichere berechnete Abweichung vor erneuter Berechnung
234  lastcalctime=millis();
235  }
236  // only for debugging
237  #ifdef DEBUG_PROGRAM_FLOW
238  Serial.print(F("Executed: tristateRegler"));
239  Serial.println(F(" ->Modul: Regler"));
240  #endif
241 }
